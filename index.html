async function fetchGrowData() { try { console.log('Starting to fetch data...'); const response = await fetch('https://calis-de-marde-production.up.railway.app/api/grows'); if (!response.ok) { throw new Error(`HTTP error! Status: ${response.status}`); } const data = await response.json(); console.log('Fetched data successfully:', data); // Update the Current Grow section const currentGrow = document.getElementById('current-grow'); currentGrow.innerHTML = `

Current Grow

Name: ${data.name}

Objective: ${data.objective || 'Not specified'}

`; // Update the Plant Health section const plantHealth = document.getElementById('plant-health'); plantHealth.innerHTML = `

Plant Health

    ${data.plants.map(plant => `
  • ${plant.strain}: ${plant.health}
    • `).join('')}
`; // Update Upcoming Tasks section const upcomingTasks = document.getElementById('upcoming-tasks'); upcomingTasks.innerHTML = `

Upcoming Tasks

No tasks available.

`; } catch (error) { console.error('Error fetching grow data:', error); } } document.addEventListener('DOMContentLoaded', fetchGrowData);
